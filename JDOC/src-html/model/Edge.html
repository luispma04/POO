001package model;
002
003/**
004 * Represents an edge between two adjacent points on the grid.
005 */
006public class Edge {
007    private final Point start;
008    private final Point end;
009    private final int cost;
010
011    /**
012     * Creates a new edge between two points with the given cost.
013     *
014     * @param start The starting point
015     * @param end The ending point
016     * @param cost The cost of traversing this edge
017     */
018    public Edge(Point start, Point end, int cost) {
019        this.start = start;
020        this.end = end;
021        this.cost = cost;
022    }
023
024    /**
025     * Gets the starting point.
026     *
027     * @return The starting point
028     */
029    public Point getStart() {
030        return start;
031    }
032
033    /**
034     * Gets the ending point.
035     *
036     * @return The ending point
037     */
038    public Point getEnd() {
039        return end;
040    }
041
042    /**
043     * Gets the cost of traversing this edge.
044     *
045     * @return The cost
046     */
047    public int getCost() {
048        return cost;
049    }
050
051    @Override
052    public boolean equals(Object obj) {
053        if (this == obj) return true;
054        if (obj == null || getClass() != obj.getClass()) return false;
055
056        Edge edge = (Edge) obj;
057        return cost == edge.cost &&
058                ((start.equals(edge.start) && end.equals(edge.end)) ||
059                        (start.equals(edge.end) && end.equals(edge.start)));
060    }
061
062    @Override
063    public int hashCode() {
064        return start.hashCode() + end.hashCode();
065    }
066
067    @Override
068    public String toString() {
069        return start + " -> " + end + " (cost: " + cost + ")";
070    }
071}
