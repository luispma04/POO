001package simulation;
002
003import model.Individual;
004
005/**
006 * Represents a death event for an individual.
007 */
008public class DeathEvent extends Event {
009    private final Individual individual;
010
011    /**
012     * Creates a new death event for the given individual at the given time.
013     *
014     * @param time The time at which the event is scheduled
015     * @param individual The individual that will die
016     */
017    public DeathEvent(double time, Individual individual) {
018        super(time);
019        this.individual = individual;
020    }
021
022    @Override
023    public void process(Simulator simulator) {
024        // Mark the individual as dead
025        individual.kill();
026    }
027
028    /**
029     * Gets the individual associated with this death event.
030     *
031     * @return The individual
032     */
033    public Individual getIndividual() {
034        return individual;
035    }
036
037    @Override
038    public String toString() {
039        return "DeathEvent{time=" + getTime() + ", individual=" + individual + "}";
040    }
041}
