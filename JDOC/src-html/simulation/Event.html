001package simulation;
002
003/**
004 * Abstract class representing an event in the simulation.
005 */
006public abstract class Event implements Comparable<Event> {
007    private final double time;
008
009    /**
010     * Creates a new event scheduled at the given time.
011     *
012     * @param time The time at which the event is scheduled
013     */
014    public Event(double time) {
015        this.time = time;
016    }
017
018    /**
019     * Gets the scheduled time of this event.
020     *
021     * @return The scheduled time
022     */
023    public double getTime() {
024        return time;
025    }
026
027    /**
028     * Processes this event in the simulation.
029     *
030     * @param simulator The simulator
031     */
032    public abstract void process(Simulator simulator);
033
034    @Override
035    public int compareTo(Event other) {
036        return Double.compare(this.time, other.time);
037    }
038}
