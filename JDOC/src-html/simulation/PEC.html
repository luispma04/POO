001package simulation;
002
003import java.util.PriorityQueue;
004import java.util.Queue;
005
006/**
007 * Pending Event Container (PEC) for managing events in the simulation.
008 */
009public class PEC {
010    private final Queue<Event> events;
011    private int processedEvents;
012
013    /**
014     * Creates a new empty PEC.
015     */
016    public PEC() {
017        this.events = new PriorityQueue<>();
018        this.processedEvents = 0;
019    }
020
021    /**
022     * Adds an event to the PEC.
023     *
024     * @param event The event to add
025     */
026    public void addEvent(Event event) {
027        events.add(event);
028    }
029
030    /**
031     * Gets the next event from the PEC without removing it.
032     *
033     * @return The next event, or null if the PEC is empty
034     */
035    public Event peekEvent() {
036        return events.peek();
037    }
038
039    /**
040     * Gets and removes the next event from the PEC.
041     *
042     * @return The next event, or null if the PEC is empty
043     */
044    public Event nextEvent() {
045        Event event = events.poll();
046        if (event != null) {
047            processedEvents++;
048        }
049        return event;
050    }
051
052    /**
053     * Checks if the PEC is empty.
054     *
055     * @return True if the PEC is empty, false otherwise
056     */
057    public boolean isEmpty() {
058        return events.isEmpty();
059    }
060
061    /**
062     * Gets the number of events currently in the PEC.
063     *
064     * @return The number of events
065     */
066    public int size() {
067        return events.size();
068    }
069
070    /**
071     * Gets the number of processed events.
072     *
073     * @return The number of processed events
074     */
075    public int getProcessedEvents() {
076        return processedEvents;
077    }
078}
