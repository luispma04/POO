001package util;
002
003import model.Grid;
004import model.Point;
005
006import java.util.ArrayList;
007import java.util.List;
008import java.util.Random;
009
010/**
011 * Utility class for generating random grid elements.
012 */
013public class RandomGenerator {
014    /**
015     * Generates random obstacles.
016     *
017     * @param numObstacles The number of obstacles to generate
018     * @param width The grid width
019     * @param height The grid height
020     * @param initialPoint The initial point
021     * @param finalPoint The final point
022     * @param random The random number generator
023     * @return A list of randomly generated obstacle points
024     */
025    public static List<Point> generateRandomObstacles(int numObstacles, int width, int height,
026                                                      Point initialPoint, Point finalPoint, Random random) {
027        List<Point> obstacles = new ArrayList<>();
028
029        while (obstacles.size() < numObstacles) {
030            int x = random.nextInt(width) + 1;
031            int y = random.nextInt(height) + 1;
032            Point obstacle = new Point(x, y);
033
034            // Don't place obstacles at the initial or final points
035            if (!obstacle.equals(initialPoint) && !obstacle.equals(finalPoint) && !obstacles.contains(obstacle)) {
036                obstacles.add(obstacle);
037            }
038        }
039
040        return obstacles;
041    }
042
043    /**
044     * Generates random special cost zones.
045     *
046     * @param numZones The number of zones to generate
047     * @param width The grid width
048     * @param height The grid height
049     * @param random The random number generator
050     * @return A list of randomly generated special cost zones
051     */
052    public static List<Grid.SpecialCostZone> generateRandomSpecialCostZones(int numZones, int width, int height, Random random) {
053        List<Grid.SpecialCostZone> zones = new ArrayList<>();
054
055        for (int i = 0; i < numZones; i++) {
056            // Generate random rectangle coordinates
057            int x1 = random.nextInt(width - 1) + 1;
058            int y1 = random.nextInt(height - 1) + 1;
059            int x2 = random.nextInt(width - x1) + x1 + 1;
060            int y2 = random.nextInt(height - y1) + y1 + 1;
061
062            // Generate random cost (between 2 and 10)
063            int cost = random.nextInt(9) + 2;
064
065            Grid.SpecialCostZone zone = new Grid.SpecialCostZone(new Point(x1, y1), new Point(x2, y2), cost);
066            zones.add(zone);
067        }
068
069        return zones;
070    }
071}
